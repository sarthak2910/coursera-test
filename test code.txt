def fibonacci(n):
    """
    Generates the first n numbers in the Fibonacci series.
    """
    fib_series = [0, 1]
    for i in range(2, n):
        next_fib = fib_series[-1] + fib_series[-2]
        fib_series.append(next_fib)
    return fib_series

# Example: Generate the first 10 numbers in the Fibonacci series
n = 10
fibonacci_series = fibonacci(n)
print(f"The first {n} numbers in the Fibonacci series are: {fibonacci_series}")
